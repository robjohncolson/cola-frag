\section{Preparation Instructions}

\begin{enumerate}
    \item Add 156 microliters Ethanol (95\%) to a 1ml amber glass vial.
    
    \item Add ingredients in the following order using separate micropipette tips:
    \begin{itemize}
        \item \textbf{Base \& Musk Foundation}: Hedione, Helvetolide, Galaxolide 27\%, Ambroxan 10\%, Iso E Super
        \item \textbf{Smoky Core}: Birch Tar, 2-Isobutyl Quinoline
        \item \textbf{Enhanced Green Complex}: \textcolor{blue}{Juniper Berry Oil 10\%}, \textcolor{blue}{Galbanum Oil 10\%}, Petitgrain Absolute, Clary Sage Absolute, \textcolor{blue}{Cypress Absolute}
        \item \textbf{Resinous Depth}: \textcolor{blue}{Cistus Absolute 23\%}
        \item \textbf{Floral Bridge}: \textcolor{blue}{Cassie Absolute 10\%}, Neroli, Lyral, Black Currant Bud Absolute
        \item \textbf{Woody-Earthy Foundation}: Patchouli EO, Patchoulol, Pink Pepper
        \item \textbf{Citrus Top}: Bergamot, Linalyl Acetate, Linalool, Coranol, \textcolor{blue}{Robertet Pineapple Essence}
        \item \textbf{Final Touches}: Veramoss, Damascone Alpha
    \end{itemize}
    
    \item Seal vial, roll gently for 1--2 minutes to blend.
    
    \item Rest for 48--72 hours in a cool, dark place before testing.
\end{enumerate}

\section{Enhanced Scent Profile}

\begin{itemize}
    \item \textbf{Opening (0-30 min)}: Sophisticated bergamot sparkle enhanced with \textcolor{blue}{juniper berry's gin-like complexity}, \textcolor{blue}{subtle pineapple signature}, and \textcolor{blue}{galbanum's sharp green bite}. Clean petitgrain bitterness bridges to the heart while black currant adds green-fruity tartness.
    
    \item \textbf{Heart (30 min - 3 hr)}: Galaxolide's clean powder foundation meets birch tar's refined smoke with \textcolor{blue}{cypress adding dry Mediterranean woody depth}. \textcolor{blue}{Cassie absolute provides honey-floral sophistication} replacing elderflower. Hedione provides radiant floral-citrus lift while massive ambroxan foundation ensures ozonic smoothness. \textcolor{blue}{Cistus adds resinous-amber complexity} with leathery-animalic facets.
    
    \item \textbf{Base (3-8+ hr)}: Galaxolide and helvetolide create lasting clean-powdery musk foundation. Patchouli provides earthy depth while birch tar maintains distinctive smoky signature. Iso E Super ensures transparent woody radiance with \textcolor{blue}{cypress adding refined coniferous sophistication}.
\end{itemize}

\section{Key Enhancements}

\begin{itemize}
    \item \textbf{Authentic Aventus Signature}: \textcolor{blue}{Robertet pineapple essence (2 μL)} provides sophisticated fruity signature at 25\% of original intensity

    \item \textbf{Removed Redundancy}: Eliminated violet leaf and boronia to focus formula and prevent powder overload

\end{itemize}
